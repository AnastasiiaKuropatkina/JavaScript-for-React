В aрифметических операциях null --> 0, undefined --> NaN


    null + 1; //1
    null / 10; //0
    null * 10; //0

    undefined + 1; //NaN
    undefined / 10 //NaN
    undefined * 10 //NaN

isFinite: null --> true; undefined --> false


    isFinite(undefined); //false
    isFinite(null); //true
