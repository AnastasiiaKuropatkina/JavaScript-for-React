Найти минимальное и максимальное значение в массиве

Solution 1. Math.min and Math.max


var myArray = [1, 5, 6, 2, 3];
var m = Math.min(...myArray);
console.log(m)

Solution 2. Math.min.apply


var myArray = [1, 5, 6, 2, 3];
var m = Math.min.apply(null, myArray);
console.log(m)

Solution 3. forEach


let arrNew = [70, 30, 40, 50, 10];
function findMin(arr) {
    if(arr.length === 0) return;

    //let min = Infinity;
    let min = arr[0];
    arr.forEach((number) => {
        if(number < min) {
            min = number;
        }
    })
    return min
}
findMax(arrNew)
function findMax(arr) {
    return Math.max(...arr);
}
findMax(arrNew)

Solution 4. reduce


let arrOfNumbers = [1, -4653, -3, 4, 5, 54, 684, 7, 8, 9, 10];

function max(list) {
  return list.reduce((a, b) => a > b ? a : b);
}
console.log(max([...arrOfNumbers]))
